root@ip-10-0-0-165:/home/ubuntu# cat /etc/nginx/sites-available/flow-wolf.org
server {
        listen [::]:80 ipv6only=on;
	server_name flow-wolf.org;
	
	allow all;        
	
	root /var/www/flow-wolf.org;
        index index.html index.htm index.php;
	
	return 301 https://$host$request_uri;	

	location /.well-known/ {
		alias /var/www/.well-known/;
	}
}
server {
   	listen [::]:443 ipv6only=on ssl;
   	server_name flow-wolf.org;
	
	allow all;
	
	root /var/www/flow-wolf.org;
	index index.php index.html index.htm;

	ssl_certificate /etc/nginx/server-all.crt;
	ssl_certificate_key /etc/nginx/server.key;
	
	ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
	ssl_prefer_server_ciphers on;
    	ssl_ciphers "ECDHE-ECDSA-AES128-GCM-SHA256 ECDHE-RSA-AES128-GCM-SHA256 DHE-RSA-AES128-SHA";
	
	ssl_dhparam /etc/nginx/dh-2048.pem;
	
	add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
	add_header Content-Security-Policy "upgrade-insecure-requests";
}
